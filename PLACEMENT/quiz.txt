
// Just Saying: Player Squad — stores player name and score

Squad Player {
    Quark name = "Unknown Warrior"
    Quark score = 0

    Doodle showScore() {
        Print("Your current score: " + score)
    }
}

// Just Saying: Question Squad — structure for each quiz question

Squad Question {
    Quark questionText
    Quark category
    Quark difficulty
    Quark correctAnswer
}

// Just Saying: QuizMaster — handles quiz flow, scoring, sarcasm


Squad QuizMaster {
    Quark questions = []
    Quark player = new Player()
    Quark selectedCategory = ""
    Quark selectedDifficulty = ""
    Quark questionCount = 0
    Quark MAX_QUESTIONS = 5

    //  Just Saying: Start the quiz

    Doodle start() {
        Print("Welcome to *Quiz Master* — where knowledge meets sarcasm!")

        player.name = Input("Enter your name, brave human: ")

        chooseCategory()
        chooseDifficulty()

        Print("Generating your quiz on " + selectedCategory + " (" + selectedDifficulty + ") ...")

        Quark pool = getFilteredQuestions()

        //  Just Saying: Shuffle logic
        SpinCycle(i = 0; i < pool.length; i = i + 1) {
            Quark r = Random(0, pool.length - 1)
            Quark temp = pool[i]
            pool[i] = pool[r]
            pool[r] = temp
        }

        //  Just Saying: Ask only first 5 questions

        SpinCycle(i = 0; i < MAX_QUESTIONS AND i < pool.length; i = i + 1) {
            askQuestion(pool[i])
            questionCount = questionCount + 1
        }

        endGame()
    }


    //  Just Saying: Category selection

    Doodle chooseCategory() {
        Print("Choose a category: Science / History / FunFacts")
        Quark cat = Input("Category: ")

        If (cat == "Science" OR cat == "History" OR cat == "FunFacts") {
            selectedCategory = cat
        } Else {
            Print("Invalid category. Defaulting to Science.")
            selectedCategory = "Science"
        }
    }

    //  Just Saying: Difficulty selection

    Doodle chooseDifficulty() {
        Print("Choose difficulty: Easy / Medium / Hard")
        Quark diff = Input("Difficulty: ")

        If (diff == "Easy" OR diff == "Medium" OR diff == "Hard") {
            selectedDifficulty = diff
        } Else {
            Print("Invalid choice. Defaulting to Easy.")
            selectedDifficulty = "Easy"
        }
    }

    //  Just Saying: Filter questions by category & difficulty

    Doodle getFilteredQuestions() {
        Quark pool = []

        SpinCycle(i = 0; i < questions.length; i = i + 1) {
            Quark q = questions[i]

            If (q.category == selectedCategory AND q.difficulty == selectedDifficulty) {
                pool.add(q)
            }
        }

        If (pool.length < 5) {
            Print("Not enough questions! Awarding pity points +10.")
            player.score = player.score + 10
        }

        Return pool
    }

    // Ask one question

    Doodle askQuestion(q) {
        Print("Q" + (questionCount + 1) + ": " + q.questionText)
        Quark ans = Input("Your answer: ")

        evaluateAnswer(q, ans)
        player.showScore()
        Print("--------------------------------")
    }

    //  Just Saying: Evaluate correctness

    Doodle evaluateAnswer(q, ans) {
        If (ans.lower() == q.correctAnswer.lower()) {
            commentOnAnswer(True)
            applyScore(q.difficulty, True)
        } Else {
            commentOnAnswer(False)
            applyScore(q.difficulty, False)
        }
    }

    //  Just Saying: Scoring logic

    Doodle applyScore(diff, isCorrect) {
        If (diff == "Easy") {
            player.score = player.score + (isCorrect ? 5 : -2)
        }
        Else If (diff == "Medium") {
            player.score = player.score + (isCorrect ? 10 : -5)
        }
        Else If (diff == "Hard") {
            player.score = player.score + (isCorrect ? 15 : -7)
        }
    }

    //  Just Saying: Sarcasm generator

    Doodle commentOnAnswer(isCorrect) {
        If (isCorrect) {
            Print("Well, someone paid attention in school!")
        } Else {
            Print("Close… if we were grading on imagination.")
        }
    }

    //  Just Saying: End of Quiz

    Doodle endGame() {
        Print("============ QUIZ OVER ============")
        Print("Final Score: " + player.score)

        If (player.score >= 60) {
            Print("Quiz Royalty has arrived!")
        }
        Else If (player.score >= 30 AND player.score < 60) {
            Print("Quiz Master in training.")
        }
        Else {
            Print("Better luck next time, genius.")
        }

        Print("Thanks for playing, " + player.name + "!")
    }
}
